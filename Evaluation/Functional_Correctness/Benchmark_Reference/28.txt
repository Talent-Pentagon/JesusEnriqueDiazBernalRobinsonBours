#include <stdio.h>

typedef struct {
    double length;
    double width;
    double height;
    double (*volume)(struct Box*);
} Box;

double getVolume(Box* b) {
    return b->length * b->width * b->height;
}

int main() {
    Box box = {2, 3, 4, getVolume};
    printf("Volume: %.2f\n", box.volume(&box));
    return 0;
}
