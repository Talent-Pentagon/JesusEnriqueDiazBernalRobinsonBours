#include <stdio.h>
#include <stdlib.h>
#include <math.h>

typedef struct {
    double (*area)(void*);
    void* data;
} Shape;

typedef struct {
    double radius;
} Circle;

typedef struct {
    double width, height;
} Rectangle;

double circleArea(void* d) {
    Circle* c = (Circle*)d;
    return M_PI * c->radius * c->radius;
}

double rectangleArea(void* d) {
    Rectangle* r = (Rectangle*)d;
    return r->width * r->height;
}

int main() {
    Circle c = {2.0};
    Rectangle r = {4.0, 5.0};

    Shape shapes[2] = {
        {circleArea, &c},
        {rectangleArea, &r}
    };

    for (int i = 0; i < 2; i++) {
        printf("Area: %.2f\n", shapes[i].area(shapes[i].data));
    }

    return 0;
}
